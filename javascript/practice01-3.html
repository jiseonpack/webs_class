3. 테이블 출력하기

1.1 for문을 이용해서 테이블을 출력하시오!

<script>
    let table = "<table border='1'>";

    for( let i=1; i<=5; i++){
        table += "<tr>";
            for( let j=1; j<=5; j++){
                table +="<td>1</td>";
            }
        table += "</tr>";    
    }
        
    table += "</table>";

    document.write(table);
</script>
